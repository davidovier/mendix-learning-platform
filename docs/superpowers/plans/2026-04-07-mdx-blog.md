```ts
// lib/blog.ts
import fs from "fs";
import path from "path";
import matter from "gray-matter";

const BLOG_DIR = path.join(process.cwd(), "content", "blog");

export interface Post {
  slug: string;
  title: string;
  description: string;
  date: string;
  category: string;
  topics: string[];
  image?: string;
  content: string;
  readingTime: number;
}

export interface CategoryCount {
  name: string;
  count: number;
}

export interface TopicCount {
  name: string;
  count: number;
}

function calculateReadingTime(content: string): number {
  const wordsPerMinute = 200;
  const words = content.trim().split(/\s+/).length;
  return Math.ceil(words / wordsPerMinute);
}

function getPostFiles(): string[] {
  if (!fs.existsSync(BLOG_DIR)) {
    return [];
  }
  return fs.readdirSync(BLOG_DIR).filter((file) => file.endsWith(".mdx"));
}

export async function getAllPosts(): Promise<Post[]> {
  const files = getPostFiles();

  const posts = files.map((filename) => {
    const slug = filename.replace(/\.mdx$/, "");
    const filePath = path.join(BLOG_DIR, filename);
    const fileContent = fs.readFileSync(filePath, "utf-8");
    const { data, content } = matter(fileContent);

    return {
      slug,
      title: data.title || slug,
      description: data.description || "",
      date: data.date || "",
      category: data.category || "Uncategorized",
      topics: data.topics || [],
      image: data.image,
      content,
      readingTime: calculateReadingTime(content),
    } as Post;
  });

  // Sort by date descending
  return posts.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());
}

export async function getPostBySlug(slug: string): Promise<Post | null> {
  const filePath = path.join(BLOG_DIR, `${slug}.mdx`);

  if (!fs.existsSync(filePath)) {
    return null;
  }

  const fileContent = fs.readFileSync(filePath, "utf-8");
  const { data, content } = matter(fileContent);

  return {
    slug,
    title: data.title || slug,
    description: data.description || "",
    date: data.date || "",
    category: data.category || "Uncategorized",
    topics: data.topics || [],
    image: data.image,
    content,
    readingTime: calculateReadingTime(content),
  };
}

export async function getPostsByCategory(category: string): Promise<Post[]> {
  const posts = await getAllPosts();
  return posts.filter((post) => post.category === category);
}

export async function getPostsByTopic(topic: string): Promise<Post[]> {
  const posts = await getAllPosts();
  return posts.filter((post) => post.topics.includes(topic));
}

export async function getRelatedPosts(post: Post, limit: number = 3): Promise<Post[]> {
  const posts = await getAllPosts();

  // Filter out current post and find related by category or overlapping topics
  const related = posts
    .filter((p) => p.slug !== post.slug)
    .map((p) => {
      let score = 0;
      if (p.category === post.category) score += 2;
      const overlappingTopics = p.topics.filter((t) => post.topics.includes(t));
      score += overlappingTopics.length;
      return { post: p, score };
    })
    .filter((item) => item.score > 0)
    .sort((a, b) => b.score - a.score)
    .slice(0, limit)
    .map((item) => item.post);

  return related;
}

export async function getCategories(): Promise<CategoryCount[]> {
  const posts = await getAllPosts();
  const counts: Record<string, number> = {};

  posts.forEach((post) => {
    counts[post.category] = (counts[post.category] || 0) + 1;
  });

  return Object.entries(counts)
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count);
}

export async function getTopics(): Promise<TopicCount[]> {
  const posts = await getAllPosts();
  const counts: Record<string, number> = {};

  posts.forEach((post) => {
    post.topics.forEach((topic) => {
      counts[topic] = (counts[topic] || 0) + 1;
    });
  });

  return Object.entries(counts)
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count);
}

export async function getAllSlugs(): Promise<string[]> {
  const files = getPostFiles();
  return files.map((file) => file.replace(/\.mdx$/, ""));
}
```

```tsx
// app/blog/page.tsx
import type { Metadata } from "next";
import { getAllPosts, getCategories, getTopics } from "@/lib/blog";
import { PostCard } from "@/components/blog/post-card";

export const metadata: Metadata = {
  title: "Blog - Mendix Certification Tips & Guides",
  description:
    "Expert tips, concept guides, and practice strategies for passing the Mendix Intermediate Developer Certification.",
  keywords: ["mendix blog", "mendix certification tips", "mendix guides"],
};

interface BlogPageProps {
  searchParams: Promise<{ category?: string; topic?: string }>;
}

export default async function BlogPage({ searchParams }: BlogPageProps) {
  const params = await searchParams;
  const allPosts = await getAllPosts();
  const categories = await getCategories();
  const topics = await getTopics();

  // Filter posts based on query params
  let posts = allPosts;
  if (params.category) {
    posts = posts.filter((post) => post.category === params.category);
  }
  if (params.topic) {
    posts = posts.filter((post) => post.topics.includes(params.topic));
  }

  const activeCategory = params.category || null;
  const activeTopic = params.topic || null;

  return (
    <div className="container mx-auto px-4 py-8">
      <div className="max-w-4xl mx-auto">
        {/* Header */}
        <div className="text-center mb-10">
          <h1 className="text-3xl font-bold tracking-tight mb-3">
            Mendix Certification Blog
          </h1>
          <p className="text-muted-foreground">
            Tips, guides, and strategies to help you pass the Mendix Intermediate exam
          </p>
        </div>

        {/* Filters */}
        <div className="mb-8 space-y-4">
          {/* Category Filter */}
          <div className="flex flex-wrap gap-2">
            <a
              href="/blog"
              className={`px-3 py-1.5 rounded-full text-sm font-medium transition-colors ${
                !activeCategory
                  ? "bg-primary text-primary-foreground"
                  : "bg-muted hover:bg-muted/80"
              }`}
            >
              All
            </a>
            {categories.map((cat) => (
              <a
                key={cat.name}
                href={`/blog?category=${encodeURIComponent(cat.name)}`}
                className={`px-3 py-1.5 rounded-full text-sm font-medium transition-colors ${
                  activeCategory === cat.name
                    ? "bg-primary text-primary-foreground"
                    : "bg-muted hover:bg-muted/80"
                }`}
              >
                {cat.name} ({cat.count})
              </a>
            ))}
          </div>

          {/* Topic Filter */}
          <div className="flex flex-wrap gap-2">
            {topics.slice(0, 8).map((topic) => (
              <a
                key={topic.name}
                href={`/blog?topic=${encodeURIComponent(topic.name)}`}
                className={`px-2.5 py-1 rounded text-xs transition-colors ${
                  activeTopic === topic.name
                    ? "bg-primary/20 text-primary"
                    : "bg-muted/50 text-muted-foreground hover:bg-muted"
                }`}
              >

              </a>

              View all posts

          </div>
        )}
      </div>
    </div>
  );
```

```tsx
// app/blog/[slug]/page.tsx
import type { Metadata } from "next";
import { notFound } from "next/navigation";
import { MDXRemote } from "next-mdx-remote/rsc";
import { getPostBySlug, getRelatedPosts, getAllSlugs } from "@/lib/blog";
import { PostHeader } from "@/components/blog/post-header";
import { PostFooter } from "@/components/blog/post-footer";
import { JsonLd, articleSchema } from "@/lib/structured-data";

interface PostPageProps {
  params: Promise<{ slug: string }>;
}

export async function generateStaticParams() {
  const slugs = await getAllSlugs();
  return slugs.map((slug) => ({ slug }));
}

export async function generateMetadata({ params }: PostPageProps): Promise<Metadata> {
  const { slug } = await params;
  const post = await getPostBySlug(slug);

  if (!post) {
    return { title: "Post Not Found" };
  }

  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL || "https://mendix-prep.vercel.app";

  return {
    title: post.title,
    description: post.description,
    keywords: [post.category, ...post.topics],
    alternates: {
      canonical: `${siteUrl}/blog/${slug}`,
    },
    openGraph: {
      type: "article",
      title: post.title,
      description: post.description,
      publishedTime: post.date,
      url: `${siteUrl}/blog/${slug}`,
    },
  };
}

export default async function PostPage({ params }: PostPageProps) {
  const { slug } = await params;
  const post = await getPostBySlug(slug);

  if (!post) {
    notFound();
  }

  const relatedPosts = await getRelatedPosts(post, 3);
  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL || "https://mendix-prep.vercel.app";

  return (
    <article className="container mx-auto px-4 py-8">
      <div className="max-w-3xl mx-auto">
        <JsonLd
          data={articleSchema({
            title: post.title,
            description: post.description,
            date: post.date,
            url: `${siteUrl}/blog/${slug}`,
          })}
        />

        <PostHeader post={post} />

        <div className="prose prose-neutral dark:prose-invert max-w-none">
          <MDXRemote source={post.content} />
        </div>

        <PostFooter post={post} relatedPosts={relatedPosts} />
      </div>
    </article>
  );
}
```

```mdx
The Mendix Intermediate Developer Certification is a significant milestone in your low-code development career. This guide covers everything you need to know to pass on your first attempt.

## Understanding the Exam Format

The Mendix Intermediate exam consists of **50 multiple-choice questions** that you must complete in **90 minutes**. You need a score of **70% or higher** to pass, which means you can miss up to 15 questions.

The exam covers six main topic areas:
- Domain Model and Data Management
- Microflows and Logic
- Nanoflows and Client-Side Logic
- Security and Access Control
- Pages and User Interface
- Integration and APIs

## Create a Study Timeline

Most candidates need **30-40 hours** of preparation time. Here's a recommended 4-week study plan:

**Week 1-2: Foundation**
- Complete Mendix Academy learning paths
- Focus on understanding concepts, not memorizing
- Take notes on areas that confuse you

**Week 3: Deep Dive**
- Study your weak areas identified in week 1-2
- Practice with real Mendix Studio Pro
- Build small test applications

**Week 4: Practice & Review**
- Take practice exams under timed conditions
- Review incorrect answers thoroughly
- Focus on understanding *why* answers are correct

## Topic Priorities

Not all topics carry equal weight. Based on exam patterns, prioritize your study time:

1. **Microflows** (High Priority) - Logic flows, activities, error handling
2. **Security** (High Priority) - Access rules, module roles, entity security
3. **Domain Model** (Medium Priority) - Associations, delete behavior, inheritance
4. **XPath** (Medium Priority) - Query syntax, constraints, tokens
5. **Pages** (Medium Priority) - Data views, list views, widgets
6. **Nanoflows** (Lower Priority) - Client-side logic, offline support

## Exam Day Tips

**Before the Exam:**
- Get a good night's sleep
- Eat a proper meal
- Test your internet connection and webcam
- Have your ID ready for verification

**During the Exam:**
- Read each question carefully - look for keywords like "BEST," "FIRST," or "NOT"
- Flag difficult questions and return to them
- Don't spend more than 2 minutes on any single question
- Trust your first instinct unless you're certain it's wrong

**Time Management:**
- 90 minutes for 50 questions = ~1.8 minutes per question
- Aim to finish with 10-15 minutes for review
- Don't panic if you're unsure - make your best guess and move on

## Practice Strategy

The most effective way to prepare is through active practice:

1. **Take timed practice exams** to simulate real conditions
2. **Review every wrong answer** - understand why the correct answer is right
3. **Identify patterns** in the types of questions you miss
4. **Build sample apps** to reinforce concepts hands-on

Ready to start practicing? Our [practice quiz](/practice) features 268 exam-style questions with instant feedback.

## Common Pitfalls to Avoid

- **Over-relying on memorization** - Understand concepts, don't just memorize facts
- **Skipping the Academy** - The official learning paths cover exactly what's tested
- **Not practicing under time pressure** - The 90-minute limit is real
- **Ignoring weak areas** - Face your difficult topics head-on

## Final Thoughts

The Mendix Intermediate Certification is achievable with proper preparation. Focus on understanding concepts deeply, practice regularly, and approach the exam with confidence.

Good luck with your certification journey!
```

```mdx
One of the most common questions on the Mendix Intermediate exam involves understanding when to use microflows versus nanoflows. This guide breaks down everything you need to know.

## What Are Microflows?

Microflows are **server-side** logic flows in Mendix. They execute on the Mendix Runtime server and are used for:

- Database operations (create, retrieve, update, delete)
- Complex business logic
- Integration with external systems
- Operations requiring access to the full database

**Key characteristics:**
- Run on the server
- Can access all data in the database
- Require a network round-trip
- Cannot run offline

## What Are Nanoflows?

Nanoflows are **client-side** logic flows that execute directly in the user's browser or on their mobile device. They're used for:

- Quick user interface updates
- Offline functionality
- Simple calculations
- Client-side validation

**Key characteristics:**
- Run in the browser/device
- Only access data already loaded on the client
- No network round-trip needed
- Essential for offline-first applications

## Key Differences

| Aspect | Microflow | Nanoflow |
|--------|-----------|----------|
| **Execution** | Server-side | Client-side |
| **Speed** | Slower (network trip) | Faster (local) |
| **Offline** | Not available | Available |
| **Data Access** | Full database | Client-loaded only |
| **Use Case** | Complex logic, DB ops | UI updates, offline |

## When to Use Each

**Use Microflows when:**
- Retrieving data from the database
- Performing commits to the database
- Calling web services or REST APIs
- Executing complex business rules
- Operations that need transaction support

**Use Nanoflows when:**
- Showing/hiding UI elements
- Simple calculations on existing objects
- Client-side validation before submit
- Building offline-capable apps
- Performance is critical for user experience

## Common Exam Mistakes

1. **Thinking nanoflows can retrieve from the database** - They can only work with objects already on the client

2. **Using microflows for simple UI logic** - This causes unnecessary server round-trips

3. **Forgetting offline implications** - If your app needs offline support, plan for nanoflows early

4. **Not understanding the performance trade-off** - Microflows are more powerful but slower

## Practice Question Example

*"A user needs to validate that an email field is properly formatted before submitting a form. Which should you use?"*

**Answer:** Nanoflow - This is client-side validation that doesn't require database access.
```

```mdx
XPath is essential for retrieving and filtering data in Mendix. This cheat sheet covers the most important syntax and functions you'll need for the certification exam.

## Basic Syntax

XPath expressions in Mendix follow this pattern:

```
//Module.Entity[constraint]
```

**Examples:**
```
//Sales.Order[Status = 'Completed']
//HR.Employee[Age > 30]
//CRM.Customer[contains(Email, '@company.com')]
```

## Comparison Operators

| Operator | Description | Example |
|----------|-------------|---------|
| `=` | Equal | `[Status = 'Active']` |
| `!=` | Not equal | `[Status != 'Deleted']` |
| `>` | Greater than | `[Amount > 1000]` |
| `<` | Less than | `[Age < 18]` |
| `>=` | Greater or equal | `[Rating >= 4]` |
| `<=` | Less or equal | `[Stock <= 10]` |

## Logical Operators

Combine multiple conditions:

```
//Sales.Order[Status = 'Pending' and Amount > 500]
//HR.Employee[Department = 'IT' or Department = 'HR']
//CRM.Customer[not(IsActive)]
```

## Common Functions

**String functions:**
```
contains(Name, 'John')      // Name contains 'John'
starts-with(Code, 'PRD')    // Code starts with 'PRD'
string-length(Name) > 10    // Name longer than 10 chars
```

**Null checks:**
```
[Email != empty]            // Email is not empty
[Manager = empty]           // Manager is empty/null
```

## Tokens (Current User/Time)

Mendix provides special tokens:

| Token | Description |
|-------|-------------|
| `[%CurrentUser%]` | The logged-in user |
| `[%CurrentDateTime%]` | Current date and time |
| `[%CurrentObject%]` | Current context object |
| `[%BeginOfCurrentDay%]` | Start of today |
| `[%EndOfCurrentDay%]` | End of today |

**Examples:**
```
//Task.Task[AssignedTo = '[%CurrentUser%]']
//Sales.Order[OrderDate > '[%BeginOfCurrentDay%]']
```

## Association Traversal

Navigate associations using `/`:

```
//Sales.OrderLine[Sales.OrderLine_Order/Sales.Order/Status = 'Shipped']
//HR.Employee[HR.Employee_Department/HR.Department/Name = 'Engineering']
```

## Exam Tips

1. **Remember the double slash** - XPath always starts with `//`
2. **Module prefix required** - Always include `Module.Entity`
3. **Quotes for strings** - String values need single quotes
4. **Case sensitivity** - XPath is case-sensitive
5. **Token syntax** - Tokens use `[% %]` brackets

## Practice Examples

Try writing XPath for these scenarios:

1. All active customers from the USA
2. Orders placed today worth over $100
3. Employees whose manager is the current user
4. Products with names containing "Pro"

**Answers:**
```
//CRM.Customer[IsActive = true and Country = 'USA']
//Sales.Order[OrderDate > '[%BeginOfCurrentDay%]' and Total > 100]
//HR.Employee[HR.Employee_Manager = '[%CurrentUser%]']
//Inventory.Product[contains(Name, 'Pro')]
```

Practice more with our [XPath study materials](/study/xpath).
```

```mdx
After analyzing hundreds of practice exam attempts, we've identified the most common mistakes that trip up Mendix Intermediate certification candidates. Avoid these pitfalls to boost your score.

## 1. Confusing Microflows and Nanoflows

**The Mistake:** Selecting microflow when nanoflow is correct (or vice versa).

**How to Avoid:** Remember the key rule - microflows run on the server and can access the database. Nanoflows run on the client and only work with data already loaded.

## 2. Forgetting Association Ownership

**The Mistake:** Not understanding which entity owns an association and how this affects delete behavior.

**How to Avoid:** The entity with the association property is the "owner." When the owner is deleted, the association reference is automatically cleaned up.

## 3. Misunderstanding Access Rules

**The Mistake:** Thinking access rules apply to all users, or confusing module roles with user roles.

**How to Avoid:** Access rules are defined per module role. A user's access is determined by which module roles their user role includes.

## 4. XPath Syntax Errors

**The Mistake:** Forgetting module prefixes, using wrong quotes, or incorrect operator syntax.

**How to Avoid:** Always use the format `//Module.Entity[constraint]`. Strings need single quotes. Practice writing XPath expressions regularly.

## 5. Ignoring Error Handling

**The Mistake:** Not considering what happens when a microflow encounters an error.

**How to Avoid:** Understand error handling activities (Error Event, Continue, Rollback). Know when transactions are rolled back.

## 6. Security Level Confusion

**The Mistake:** Not knowing the difference between Prototype/Demo and Production security levels.

**How to Avoid:** Prototype/Demo mode allows all access by default. Production mode denies everything not explicitly allowed.

## 7. Misreading Questions

**The Mistake:** Missing keywords like "BEST," "FIRST," "NOT," or "EXCEPT."

**How to Avoid:** Read each question twice. Circle or note keywords. "Which is NOT true" means 3 answers ARE true.

## 8. Overthinking Simple Questions

**The Mistake:** Looking for tricks in straightforward questions.

**How to Avoid:** Sometimes the obvious answer is correct. If a question seems simple, it probably is.

## 9. Poor Time Management

**The Mistake:** Spending too long on difficult questions and running out of time.

**How to Avoid:** Set a pace of ~1.5 minutes per question. Flag hard questions and return to them. Never leave questions blank.

## 10. Not Reviewing Flagged Questions

**The Mistake:** Running out of time without reviewing uncertain answers.

**How to Avoid:** Aim to finish with 10-15 minutes left. Your second look often catches mistakes.

## Bonus Tips

- **Read all options** before selecting - the "best" answer might be option D
- **Eliminate obviously wrong answers** first
- **Trust your preparation** - you've studied, so trust your knowledge
- **Stay calm** - anxiety causes careless mistakes

Ready to avoid these mistakes in practice? Take our [practice exam](/exam) to test yourself under real conditions.
```

```mdx
Security is one of the most heavily tested topics on the Mendix Intermediate exam. This guide explains how Mendix security works from the ground up.

## Security Model Overview

Mendix security operates on multiple levels:

1. **Project Security** - Overall security settings
2. **Module Security** - Access rules per module
3. **Entity Security** - Who can access what data
4. **Page Security** - Who can view which pages
5. **Microflow Security** - Who can execute which logic

## Security Levels

Mendix has two security levels:

**Prototype/Demo Mode:**
- Used during development
- All access allowed by default
- Easy to test functionality
- Never use in production!

**Production Mode:**
- Used for live applications
- All access denied by default
- Must explicitly grant permissions
- Required for deployment

## Module Roles vs User Roles

This distinction trips up many candidates:

**Module Roles:**
- Defined within a module
- Determine access to module content
- Examples: "OrderManager," "OrderViewer"

**User Roles:**
- Defined at project level
- Assigned to actual users
- Combine multiple module roles
- Examples: "Administrator," "SalesRep"

A user role like "SalesRep" might include:
- "OrderManager" from the Sales module
- "CustomerViewer" from the CRM module
- "ReportViewer" from the Reporting module

## Access Rules

Access rules determine what users can do with entities:

**Rule Components:**
- **Module Role** - Which role this rule applies to
- **Access Rights** - Create, Read, Write, Delete
- **XPath Constraint** - Limit which objects are accessible
- **Member Access** - Which attributes/associations are accessible

**Example:**
A "SalesRep" module role might have this access to Orders:
- Create: Yes
- Read: Yes (with XPath: `[SalesRep = '[%CurrentUser%]']`)
- Write: Yes (only their own orders)
- Delete: No

## XPath Constraints in Security

XPath constraints limit which objects a user can access:

```
[Owner = '[%CurrentUser%]']           // Only own records
[Department = '[%CurrentUser%]/Department']  // Same department
[Status != 'Confidential']            // Not confidential
```

**Important:** XPath constraints apply to ALL access - retrieve, change, delete.

## Entity Access Summary

| Access Type | Description |
|-------------|-------------|
| Create | Can create new objects |
| Read | Can retrieve/view objects |
| Write | Can modify existing objects |
| Delete | Can delete objects |

Each can be:
- **None** - No access
- **Yes** - Full access (for this role)
- **XPath** - Conditional access based on constraint

## Page and Microflow Security

**Pages:**
- Linked to module roles
- If no role has access, page is hidden in navigation
- Users without access see "Access Denied"

**Microflows:**
- Also linked to module roles
- Can be set to "Allowed" or specific roles
- Buttons calling restricted microflows are hidden

## Common Exam Questions

1. *"What happens when a user tries to access data outside their XPath constraint?"*
   - The data is simply not returned - no error shown

2. *"In Production mode, what is the default access?"*
   - All access denied - must be explicitly granted

3. *"Can XPath constraints reference the current user?"*
   - Yes, using `[%CurrentUser%]` token

## Security Best Practices

1. **Start with Production mode** - Even in development
2. **Least privilege principle** - Only grant necessary access
3. **Test all roles** - Use demo users for each role
4. **Document your rules** - Complex security needs documentation
5. **Review regularly** - Security requirements change

Solidify your understanding with our [security practice questions](/study/security).
```

```ts
import type { MetadataRoute } from "next";
import { topics } from "@/lib/content/topics";
import { getAllPosts } from "@/lib/blog";

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL || "https://mendix-prep.vercel.app";

  const staticPages = [
    "",
    "/study",
    "/practice",
    "/exam",
    "/cheatsheet",
    "/progress",
    "/pricing",
    "/blog",
    "/login",
    "/signup",
  ];

  const topicPages = topics.map((topic) => `/study/${topic.id}`);

  // Get blog posts
  const posts = await getAllPosts();
  const blogPages = posts.map((post) => `/blog/${post.slug}`);

  const allPages = [...staticPages, ...topicPages, ...blogPages];

  return allPages.map((route) => ({
    url: `${siteUrl}${route}`,
    lastModified: new Date(),
    changeFrequency: route === "" ? "weekly" : route.startsWith("/blog/") ? "monthly" : "monthly",
    priority: route === "" ? 1 : route === "/blog" ? 0.8 : route.startsWith("/blog/") ? 0.6 : 0.8,
  }));
}
```
